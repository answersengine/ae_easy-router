def initialize_hook_router_plugin_router opts = {}
  opts = {
    config: nil,
    config_file_path: nil,
    force: false
  }.merge opts
  @local_config = opts[:config]
  @local_config ||= AeEasy::Config::Local.new(
    file_path: opts[:config_file_path],
    force: opts[:force]
  )
end
